import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
   /* Citit un caracter si 2 numere de la tastatura, si stocati-le in variabile.
   Folosind structura switch, daca s-a introdus un caracter anume,
   realizati operatia respectiva cu cele 2 numere, si afisati rezultatul pe ecran:
        -‘a’: adaugare
        -‘s’: scadere
        -‘i’: inmultire
        -‘p’: impartire
        -‘m’: modul*/

        Scanner scanner = new Scanner(System.in);
        String caracter = scanner.nextLine();
        double numar1 = scanner.nextDouble();
        double numar2 = scanner.nextDouble();

        switch (caracter){
            case "+":
                System.out.println("'a' " + (numar1+numar2));
                break;
            case "-":
                System.out.println("'s' " + (numar1-numar2));
                break;
            case "*":
                System.out.println("'i' " + (numar1*numar2));
                break;
            case "/":
                System.out.println("'p' " + (numar1/numar2));
                break;
            case "%":
                System.out.println("'m' " + (numar1%numar2));
                break;
            default:
                System.out.println("Caracter diferit de operator matematic");

        }
    }
}
